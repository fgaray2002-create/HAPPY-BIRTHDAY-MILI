    <script>
        $(document).ready(function() {
            // Inicializar el Flipbook
            $("#libro").turn({
                width: 800,
                height: 500,
                autoCenter: true
           });

           // Evento cuando cambia la página
           $("#libro").bind("turned", function(event, page, view) {
               let audio = document.getElementById('reproductor');
               let nuevaMusica = "";

               // --- AQUÍ DEFINÍS QUÉ CANCIÓN SUENA EN QUÉ PÁGINA ---
               if (page >= 1 && page <= 3) {
                   nuevaMusica = "1.mp3"; // El audio 1 suena de la pág 1 a la 3
               } else if (page == 4) {
                   nuevaMusica = "2.mp3"; // El audio 2 suena en la pág 4
               } else if (page >= 5 && page <= 7) {
                   nuevaMusica = "3.mp3"; // El audio 3 suena de la 5 a la 7
               } else {
                   nuevaMusica = page + ".mp3"; // El resto sigue el orden normal
               }

               // Solo cambia el audio si es una canción distinta a la que ya está sonando
               let nombreArchivoActual = audio.src.split('/').pop();
               if (nombreArchivoActual !== nuevaMusica) {
                   audio.src = nuevaMusica;
                   audio.play().catch(e => console.log("Clickeá para activar sonido"));
               }
          });
    });
</script>
